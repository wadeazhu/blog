vue-quill-editor: 3.0.6

## 代码

```vue
<template>
  <div class="my-editor">
    <quill-editor
      class="ql-editor"
      v-if="!disabled"
      v-model="content"
      ref="myQuillEditor"
      :options="editorOption"
      :disabled="disabled"
      @blur="onEditorBlur($event)"
      @focus="onEditorFocus($event)"
      @change="onEditorChange($event)"
      @ready="onEditorReady($event)"
    >
    </quill-editor>
    <p v-if="!disabled" class="count">{{ `${tiLength}/${maximum}` }}</p>

    <div v-else v-html="value"></div>

    <Upload v-show="false" class="avatar-uploader" :value="uploadFile" @editor-image="getUploadList"></Upload>
  </div>
</template>

<script>
import Upload from '@/components/common/upload.vue'
export default {
  components: { Upload },
  props: {
    value: {
      type: String,
      default: ''
    },
    // 限制最大数
    maximum: {
      type: Number,
      default: 3000
    },
    disabled: {
      type: Boolean,
      default: false
    }
  },
  data () {
    return {
      editor: null, // 富文本编辑器对象
      content: this.value, // 富文本编辑器默认内容
      uploadFile: [],
      tiLength: 0, // 当前字数
      editorOption: {
        modules: {
          toolbar: {
            container: [
              ['bold', 'italic', 'underline', 'strike'], // 加粗 斜体 下划线 删除线
              ['blockquote', 'code-block'], // 引用  代码块
              [{ header: 1 }, { header: 2 }], // 1、2 级标题
              [{ list: 'ordered' }, { list: 'bullet' }], // 有序、无序列表
              [{ script: 'sub' }, { script: 'super' }], // 上标/下标
              [{ indent: '-1' }, { indent: '+1' }], // 缩进
              [{ direction: 'rtl' }], // 文本方向
              // [{ size: ['12', '14', '16', '18', '20', '22', '24', '28', '32', '36'] }], // 字体大小 失效
              [{ size: ['small', false, 'large', 'huge'] }],
              [{ header: [1, 2, 3, 4, 5, 6] }], // 标题
              [{ color: [] }, { background: [] }], // 字体颜色、字体背景颜色
              // [{ font: ['songti'] }], // 字体种类
              [{ align: [] }], // 对齐方式
              ['clean'], // 清除文本格式
              ['image'] // 链接、图片、视频
            ],
            handlers: {
              'image': value => {
                 // 图片上传时触发上传组件
                if (value && this.tiLength < this.maximum) {
                  document.querySelector('.avatar-uploader input').click()
                } else {
                  // this.quill.format('image', false)
                  this.$Notice.warning({ title: '内容已超出限制' })
                }
              }
            }
          }
        },
        placeholder: '请输入内容'
      }
    }
  },
  watch: {
    value (val) {
      this.content = val
      this.tiLength = this.$refs.myQuillEditor.quill.getLength() - 1
    }
  },
  beforeDestroy () {
    this.editor = null
    delete this.editor
  },
  methods: {
    // 失去焦点事件
    onEditorBlur (quill) {
      // console.log('editor blur!', quill)
      // this.$emit('change-content', this.content)
    },
    // 获得焦点事件
    onEditorFocus ({ quill }) {
      // quill.enable(true)
      // console.log('editor focus!', quill)
    },
    // 准备富文本编辑器
    onEditorReady (quill) {
      // console.log('editor ready!', quill)
    },
    // 内容改变事件
    onEditorChange ({ quill, html, text }) {
      let e = this.$refs.myQuillEditor

      // 处理字数限制
      if (e.quill.getLength() - 1 > this.maximum) {
        e.quill.deleteText(this.maximum, 2)
        quill.enable(false)
        this.$nextTick(() => quill.enable(true))
        return
      }
      // 显示当前字数
      if (this.content == '') this.tiLength = 0
      else this.tiLength = e.quill.getLength() - 1

      this.content = html
      // 判断空标签并提交数据
      if (!this.checkVal(html).trim()) this.$emit('change-content', '')
      else this.$emit('change-content', html)
    },

    checkVal (str) {
      return str
        .replace(/<[^<p>]+>/g, '') // 将所有<p>标签 replace ''
        .replace(/<[</p>$]+>/g, '') // 将所有</p>标签 replace ''
        .replace(/&nbsp;/gi, '') // 将所有 空格 replace ''
        .replace(/<[^<br/>]+>/g, '') // 将所有 换行符 replace ''
    },
    // 图片处理
    getUploadList (url) {
      const imageUrl = url
      // 获取光标所在位置
      let quill = this.$refs.myQuillEditor.quill
      let length = quill.getSelection().index
      // 插入图片
      quill.insertEmbed(length, 'image', imageUrl)
      // 调整光标到最后
      quill.setSelection(length + 1)
    }
  }
}
</script>
```
